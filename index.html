import pygame import random # --- CONFIGURAÇÕES --- WIDTH, HEIGHT = 600, 800 LANE_WIDTH = WIDTH // 3 FPS = 60 class Player(pygame.sprite.Sprite): def __init__(self): super().__init__() # Carregando os frames de animação try: self.frames = [ pygame.image.load("player_run1.png").convert_alpha(), pygame.image.load("player_run2.png").convert_alpha() ] except: # Fallback caso não existam as imagens ainda self.frames = [pygame.Surface((70, 110)) for _ in range(2)] self.frames[0].fill((0, 255, 128)) self.frames[1].fill((0, 200, 100)) self.current_frame = 0 self.image = self.frames[self.current_frame] self.rect = self.image.get_rect() self.lane = 1 self.rect.bottom = HEIGHT - 50 self.update_lane() def update_lane(self): self.rect.centerx = self.lane * LANE_WIDTH + (LANE_WIDTH // 2) def update(self): # Animação simples: alterna entre os frames self.current_frame += 0.1 if self.current_frame >= len(self.frames): self.current_frame = 0 self.image = self.frames[int(self.current_frame)] class Obstacle(pygame.sprite.Sprite): def __init__(self, speed): super().__init__() try: self.image = pygame.image.load("obstacle.png").convert_alpha() except: self.image = pygame.Surface((LANE_WIDTH - 20, 150)) self.image.fill((255, 50, 50)) self.rect = self.image.get_rect() self.rect.centerx = random.randint(0, 2) * LANE_WIDTH + (LANE_WIDTH // 2) self.rect.bottom = 0 self.speed = speed def update(self): self.rect.y += self.speed if self.rect.top > HEIGHT: self.kill() def main(): pygame.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) clock = pygame.time.Clock() # --- LÓGICA DO FUNDO INFINITO --- try: bg = pygame.image.load("background.png").convert() bg = pygame.transform.scale(bg, (WIDTH, HEIGHT)) except: bg = pygame.Surface((WIDTH, HEIGHT)) bg.fill((30, 30, 30)) # Cinza escuro se não tiver imagem bg_y = 0 player = Player() all_sprites = pygame.sprite.Group(player) obstacles = pygame.sprite.Group() game_speed = 10 score = 0 running = True while running: # 1. Movimentação do Fundo (O segredo do movimento infinito) # Desenhamos a imagem duas vezes, uma em cima da outra bg_y += game_speed if bg_y >= HEIGHT: bg_y = 0 screen.blit(bg, (0, bg_y)) screen.blit(bg, (0, bg_y - HEIGHT)) # 2. Eventos for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.KEYDOWN: if event.key == pygame.K_LEFT and player.lane > 0: player.lane -= 1 if event.key == pygame.K_RIGHT and player.lane < 2: player.lane += 1 player.update_lane() # 3. Criar Obstáculos if random.random() < 0.02: obs = Obstacle(game_speed) obstacles.add(obs) all_sprites.add(obs) # 4. Atualizar tudo all_sprites.update() # 5. Colisão if pygame.sprite.spritecollide(player, obstacles, False): print("Game Over!") running = False # 6. Desenhar Sprites e Pontuação all_sprites.draw(screen) score += 1 game_speed += 0.002 # O jogo fica mais rápido aos poucos pygame.display.flip() clock.tick(FPS) pygame.quit() if __name__ == "__main__": main()
